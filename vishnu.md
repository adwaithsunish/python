## SAMPLE CODE

```
import numpy as np
 arr1=np.array([1,2,3])
 print("first array\n",arr1)
 arr2=np.array([4,5,6])
 print("second array\n",arr2)
 arr=np.concatenate((arr1,arr2))
print("concatenated array\n",arr)
 newarr=np.array_split(arr,3)
 print(newarr)
 x = np.where(arr==4)
 print("found at:",x,"position")

```
